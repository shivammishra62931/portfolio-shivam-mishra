import pandas as pd
import matplotlib.pyplot as plt

portfolio = pd.read_csv('business_data.csv')
clean = portfolio.dropna()

profit = clean.groupby('month')['profit'].sum()
profit.plot(kind='line', marker='o')

plt.title('Business Profit Trend')
plt.xlabel('Month')
plt.ylabel('Profit')
plt.show()
